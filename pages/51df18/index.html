# 高阶组件 Higher order components (HOC)

HOC 是一个函数,接受一个组件返回一个新组件. 主要是抽离状态，将重复的受控组件的逻辑抽离到高阶组件中，以新的props传给受控组件中，高阶组件中可以操作props传入受控组件。开源库中常见的高阶组件：Redux的connect， react-router的withRouter等等。

function Child(prop) {
  return <div>Child</div>
}

const foo = Cmp = props => {
  return (
    <div className={border}>
      <Cmp {...props}></Cmp>
    </div>
  )
}

import React, { Component } from "react";

function FromCreate(Cmp) {
  return class extends Component {
    constructor(props){
      super()
      this.state ={}
    }
    inputChange = (e) => {
      const { name, value} = e.target
      this.setState({
        [name]: value
      })
    }
    getFieldDecorator = (filed, option) => {
      return Inputel => {
        return React.cloneElement(Inputel, {
          name: filed,
          value: this.state[filed] || '',
          onChange: this.inputChange
        })
      }
    }
    validateField = (call) => {
      let errors = {}
      Object.keys(this.state).forEach(item => {
        if(!this.state[item]){
          errors[item] ='error'
        }
      })
      return call(errors, this.state)
    }
    getFieldValues = () => {
      return {  ...this.state }
    }
    getFieldValue = (filed) => {
      return this.state[filed] || ''
    }
    render () {
      return (
        <div className="border">
          <Cmp
            getFieldDecorator={this.getFieldDecorator}
            getFieldValues={this.getFieldValues}
            getFieldValue={this.getFieldValue}
          />
        </div>
      )
    }
  }
}

export default FromCreate;
